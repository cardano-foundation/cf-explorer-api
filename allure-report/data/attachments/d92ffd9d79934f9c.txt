//Scenario Compare api: Get current epoch with api: Get all epoch
//all variable get from API api/v1/epochs/current(Get current epoch)
pm.test("Compare value between current epoch and Get all epoch", function () {
    //Main compare
    pm.expect(jsonData.data[0].no).to.eql(pm.environment.get("epoch_no"));
    pm.expect(jsonData.data[0].status).to.include("IN_PROGRESS");
    pm.expect(jsonData.data[0].maxSlot).to.eql(pm.environment.get("epoch_totalSlot"));
    pm.expect(jsonData.data[0].account).to.eql(pm.environment.get("epoch_account"));
    pm.expect(jsonData.data[0].startTime).to.eql(pm.environment.get("epoch_startTime"));
    pm.expect(jsonData.data[0].endTime).to.eql(pm.environment.get("epoch_endTime"));
});
