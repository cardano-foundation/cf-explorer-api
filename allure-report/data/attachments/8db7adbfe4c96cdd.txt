var jsonData = pm.response.json();
var stt = 200
var res_time = 5000

//Compare static data
pm.test("Status code is 200", function () {
    pm.response.to.have.status(stt);
});

pm.test("Response time is less than 5000ms", function () {
    pm.expect(pm.response.responseTime).to.be.below(res_time);
});

//Compare format data type
pm.test("Check data type of the response", () => {
    let moment = require ('moment');
    pm.expect(jsonData.txHash).to.be.a("string");
    pm.expect(jsonData.blockNo).to.be.a("number");
    pm.expect(jsonData.epoch).to.be.a("number");
    pm.expect(jsonData.outSum).to.be.a("number");
    pm.expect(jsonData.fee).to.be.a("number");
    pm.expect(jsonData.poolId).to.be.a("string");
    pm.expect(typeof jsonData.poolName).to.be.oneOf([ 'object', 'string']);
    pm.expect(jsonData.time).to.include(moment(jsonData.time).format('YYYY/MM/DD HH:mm:ss'));
    pm.expect(jsonData.stakeTotalAmount).to.be.a("number");
});

//Scenario compare get Delegation Detail with get Delegations
//all variable get from API api/v1/stake-lifecycle/:stakeKey/delegations(get Delegations)
pm.test("Check txHash value", function () {
    pm.expect(jsonData.txHash).to.include(pm.environment.get("getDelegations_data_txHash"));
});
pm.test("Check outSum value", function () {
    pm.expect(jsonData.outSum).to.eql(pm.environment.get("getDelegations_data_outSum"));
});
pm.test("Check fee value", function () {
    pm.expect(jsonData.fee).to.eql(pm.environment.get("getDelegations_data_fee"));
});
pm.test("Check time value", function () {
    pm.expect(jsonData.time).to.contain(pm.environment.get("getDelegations_data_time"));
});
