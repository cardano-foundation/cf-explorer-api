var jsonData = pm.response.json();
var stt = 200
var res_time = 5000

//Compare static data
pm.test("Status code is 200", function () {
    pm.response.to.have.status(stt);
});

pm.test("Response time is less than 5000ms", function () {
    pm.expect(pm.response.responseTime).to.be.below(res_time);
});

//Compare format data type
pm.test("Check data type of the response", () => {
    let moment = require ('moment');
    pm.expect(jsonData.txHash).to.be.a("string");
    pm.expect(jsonData.amount).to.be.a("number");
    pm.expect(jsonData.fee).to.be.a("number");
    pm.expect(jsonData.time).to.include(moment(jsonData.time).format('YYYY/MM/DD HH:mm:ss'));
    pm.expect(jsonData.stakeTotalAmount).to.be.a("number");
    pm.expect(jsonData.stakeRewardAvailable).to.be.a("number");
});

//Scenario compare get Detail Withdrawal api with get Withdrawals api
//all variable get from API api/v1/stake-lifecycle/:stakeKey/withdrawals(get Withdrawals)
pm.test("Check txHash value", function () {
    pm.expect(jsonData.txHash).to.include(pm.environment.get("getWithdrawals_data_txHash"));
});
pm.test("Check amount value", function () {
    pm.expect(jsonData.amount).to.eql(pm.environment.get("getWithdrawals_data_value"));
});
pm.test("Check fee value", function () {
    pm.expect(jsonData.fee).to.eql(pm.environment.get("getWithdrawals_data_fee"));
});
pm.test("Check time value", function () {
    pm.expect(jsonData.time).to.contain(pm.environment.get("getWithdrawals_data_time"));
});
