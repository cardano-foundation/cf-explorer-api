let pages = pm.environment.get("pages");
let sizes = pm.environment.get("sizes");

if(!pages || pages.length == 0) {
    pages = [0, 5, 10, 15];
};

if(!sizes || sizes.length == 0) {
    sizes = [5, 10, 15, 20];
};

let currentPage = pages.shift();
let currentsize = sizes.shift();

pm.environment.set("page", currentPage);
pm.environment.set("pages", pages);

pm.environment.set("size", currentsize);
pm.environment.set("sizes", sizes);
