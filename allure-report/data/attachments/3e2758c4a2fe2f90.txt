var jsonData = pm.response.json();

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Response time is less than 5000ms", function () {
    pm.expect(pm.response.responseTime).to.be.below(5000);
});

//Compare data type
pm.test("Check data type of date", () => {
    let moment = require ('moment');
  pm.expect(jsonData[0].date).to.include(moment(jsonData[0].date).format('YYYY/MM/DD HH:mm:ss'));
});

pm.test
("Check data type of simpleTransactions", () => {
  pm.expect(jsonData[0].simpleTransactions).to.be.a("number");
});

pm.test("Check data type of smartContract", () => {
  pm.expect(jsonData[0].smartContract).to.be.a("number");
});

pm.test("Check data type of metadata", () => {
  pm.expect(jsonData[0].metadata).to.be.a("number");
});
