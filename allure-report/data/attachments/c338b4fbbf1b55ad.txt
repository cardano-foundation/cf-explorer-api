var jsonData = pm.response.json();
var stt = 200
var res_time = 5000

//Compare static data
pm.test("Status code is 200", function () {
    pm.response.to.have.status(stt);
});

pm.test("Response time is less than 5000ms", function () {
    pm.expect(pm.response.responseTime).to.be.below(res_time);
});

pm.test("Check currentPage value", function () {
    pm.expect(jsonData.currentPage).to.eql(0);
});

//Compare format data type
pm.test("Check data type of the response", () => {
    let moment = require ('moment');
    pm.expect(jsonData).to.be.an("object");
    pm.expect(jsonData.totalItems).to.be.a("number");
    pm.expect(jsonData.totalPages).to.be.a("number");
    pm.expect(jsonData.currentPage).to.be.a("number");
    pm.expect(jsonData.data).to.be.an("array");
    pm.expect(jsonData.data[0].txHash).to.be.a("string");
    pm.expect(jsonData.data[0].fee).to.be.a("number");
    pm.expect(jsonData.data[0].deposit).to.be.a("number");
    pm.expect(jsonData.data[0].time).to.include(moment(jsonData.data[0].time).format('YYYY/MM/DD HH:mm:ss'));
});

//Scenario compare Get  stake deregistrations  by report id api with get Stake De Registrations api
//all variable get from API api/v1/staking-lifecycle/report/stake-key/:reportId/registrations(Get stake registrations by report id)
pm.test("Others value exist",function (){
    var tx_hash = pm.environment.get("stakingregistrations_data_txHash");
    let result = jsonData.data.find(a => a.txHash === tx_hash);
    if (result?.txHash === tx_hash){
        pm.expect(result.txHash).to.be.include(pm.environment.get("stakingregistrations_data_txHash"));
        pm.expect(result.fee).to.be.eql(pm.environment.get("stakingregistrations_data_fee"));
        pm.expect(result.deposit).to.be.eql(pm.environment.get("stakingregistrations_data_deposit"));
        pm.expect(result.time).to.be.contain(pm.environment.get("stakingregistrations_data_time"));
    }
});

//Scenario Compare api get Stake Registrations with api Get stake key registration transaction detail
pm.test("Get Randome value of get Registrations api",function(){
    //Randome value
    var randomIndex = Math.floor(Math.random() * (jsonData.data.length-1));
    pm.environment.set("getRegistrations_data_txHash", jsonData.data[randomIndex].txHash);
    pm.environment.set("getRegistrations_data_fee", jsonData.data[randomIndex].fee);
    pm.environment.set("getRegistrations_data_deposit", jsonData.data[randomIndex].deposit);
    pm.environment.set("getRegistrations_data_time", jsonData.data[randomIndex].time);
});

// Check default size 
if (jsonData.totalItems > 0){
    pm.test("Check Size default", function(){
        pm.expect(jsonData.data.length).to.be.within(1, 100);
    });
};
